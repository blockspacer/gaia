172 template <typename Req, typename Resp>
173 auto HttpsClient::Send(const Req& req, Resp* resp) -> error_code {
174  namespace h2 = ::boost::beast::http;
175  error_code ec = Send(req);
176  if (ec)
177  return ec;
178 
179  h2::read(*client_, tmp_buffer_, *resp, ec);
180  return HandleError(ec);
181 }

183 template <typename Req> auto HttpsClient::Send(const Req& req) -> error_code {
184  error_code ec = ReconnectIfNeeded();
185  if (ec)
186  return ec;
187  ::boost::beast::http::write(*client_, req, ec);
188  return HandleError(ec);

199 template <typename Parser> auto HttpsClient::Read(Parser* parser) -> error_code {
200  error_code ec;
201 
202  // Note that read returns number of raw bytes read from stream before parsing which
203  // does not correlate to the final data stored inside the parser.
204  ::boost::beast::http::read(*client_, tmp_buffer_, *parser, ec);

208 inline auto HttpsClient::HandleError(const error_code& ec) -> error_code {
209  using err = ::boost::beast::http::error;
210  if (ec && ec != err::need_buffer && ec != err::partial_message)
211  reconnect_needed_ = true;
212  return ec;
213 }
