210 bool WriteHandler::Append(strings::ByteRange* src) {
211  size_t prepare_size = std::min(src->size(), body_mb.max_size() - body_mb.size());
212  auto mbs = body_mb.prepare(prepare_size);
213  size_t offs = 0;
214  for (auto mb : mbs) {
215  memcpy(mb.data(), src->data() + offs, mb.size());
216  offs += mb.size();
217  }
218  CHECK_EQ(offs, prepare_size);
219  src->remove_prefix(prepare_size);
220  body_mb.commit(prepare_size);
221 
222  DCHECK_LE(body_mb.size(), body_mb.max_size());
223  return body_mb.size() == body_mb.max_size();
224 }

256 auto HttpsClient::DrainResponse(h2::response_parser<h2::buffer_body>* parser) -> error_code {
257  constexpr size_t kBufSize = 1 << 16;
258  std::unique_ptr<uint8_t[]> buf(new uint8_t[kBufSize]);
259  auto& body = parser->get().body();
260  error_code ec;
261  size_t sz = 0;
262 
263  // Drain pending response completely to allow reusing the current connection.
264  VLOG(1) << "parser: " << parser->got_some();
265  while (!parser->is_done()) {
266  body.data = buf.get();
267  body.size = kBufSize;
268  size_t raw_bytes = h2::read(*client_, tmp_buffer_, *parser, ec);
269  if (ec && ec != h2::error::need_buffer) {
270  VLOG(1) << "Error " << ec << "/" << ec.message();
271  return ec;
272  }
273  sz += raw_bytes;
274 
275  VLOG(1) << "DrainResp: " << raw_bytes << "/" << body.size << ", " << parser->get();
276  }
277  VLOG_IF(1, sz > 0) << "Drained " << sz << " bytes";
278 
279  return error_code{};
280 }

331  RETURN_IF_ERROR(PrepareConnection());
332 
333  string url = absl::StrCat("/storage/v1/b?project=", gce_.project_id());
334  absl::StrAppend(&url, "&fields=items,nextPageToken");
335 
336  auto http_req = PrepareRequest(h2::verb::get, url, access_token_header_);
337 
338  // TODO: to have a handler extracting what we need.
339  rj::Document doc;
340  vector<string> results;
341 
342  while (true) {
343  h2::response<h2::dynamic_body> resp_msg;
344 
345  RETURN_IF_ERROR(HttpMessage(&http_req, &resp_msg));
346  if (resp_msg.result() != h2::status::ok) {
347  return HttpError(resp_msg);
348  }
349 
350  http::RjBufSequenceStream is(resp_msg.body().data());
351 
352  doc.ParseStream<rj::kParseDefaultFlags>(is);
353  if (doc.HasParseError()) {
354  LOG(ERROR) << rj::GetParseError_En(doc.GetParseError()) << resp_msg;
355  return Status(StatusCode::PARSE_ERROR, "Could not parse json response");
356  }
357 
358  auto it = doc.FindMember("items");
359  CHECK(it != doc.MemberEnd()) << resp_msg;
360  const auto& val = it->value;
361  CHECK(val.IsArray());
362  auto array = val.GetArray();
363 
364  for (size_t i = 0; i < array.Size(); ++i) {
365  const auto& item = array[i];
366  auto it = item.FindMember("name");
367  if (it != item.MemberEnd()) {
368  results.emplace_back(it->value.GetString(), it->value.GetStringLength());
369  }
370  }
371  it = doc.FindMember("nextPageToken");
372  if (it == doc.MemberEnd()) {
373  break;
374  }
375  absl::string_view page_token{it->value.GetString(), it->value.GetStringLength()};
376  http_req.target(absl::StrCat(url, "&pageToken=", page_token));
377  }

467 StatusObject<bool> GCS::SendRequestIterative(Request* req, Parser<h2::buffer_body>* parser) {
468  VLOG(1) << "Req: " << *req;
469 
470  error_code ec = https_client_->Send(*req);
471  RETURN_EC_STATUS(ec);
472 
473  ec = https_client_->ReadHeader(parser);
474 
475  if (ec) {
476  LOG(ERROR) << "Socket error: " << ec << "/" << ec.message();
477  VLOG(1) << "FiberSocket status: " << https_client_->client()->next_layer().status();
478 
479  return false; // need to retry
480  }
481 
482  const auto& msg = parser->get();
483  VLOG(1) << "HeaderResp: " << msg;
484  if (!parser->keep_alive()) {
485  https_client_->schedule_reconnect();
486  LOG(INFO) << "Scheduling reconnect due to conn-close header";
487  }
488 
489  // Partial content can appear because of the previous reconnect.
490  if (msg.result() == h2::status::ok || msg.result() == h2::status::partial_content) {
491  return true; // all is good.
492  }
493 
494  if (ShouldRetry(msg.result())) {
495  RETURN_EC_STATUS(https_client_->DrainResponse(parser));
496  this_fiber::sleep_for(1s);
497  return false; // retry
498  }
499 
500  if (IsUnauthorized(msg)) {
501  RETURN_EC_STATUS(https_client_->DrainResponse(parser));
502 
503  auto st = RefreshToken(req);
504  if (!st.ok())
505  return st;
506  return false;
507  }
508 
509  LOG(ERROR) << "Unexpected status " << msg;
510 
511  return Status(StatusCode::INTERNAL_ERROR, "Unexpected http response");
512 }

535 auto GCS::ReadSequential(const strings::MutableByteRange& range) -> ReadObjectResult {
536  SeqReadHandler* handler = absl::get_if<SeqReadHandler>(conn_state_.get());
537  CHECK(handler && https_client_);
538 
539  if (handler->parser->is_done()) {
540  return 0;
541  }
542 
543  for (unsigned iters = 0; iters < 3; ++iters) {
544  auto& body = handler->parser->get().body();
545  body.data = range.data();
546  auto& left_available = body.size;
547  left_available = range.size();
548 
549  error_code ec = https_client_->Read(&handler->parser.value());
550  if (!ec || ec == h2::error::need_buffer || ec == h2::error::partial_message) {
551  size_t http_read = range.size() - left_available;
552  DVLOG(2) << "Read " << http_read << " bytes from " << handler->offset << " with capacity "
553  << range.size();
554 
555  // This check does not happen. See here why: https://github.com/boostorg/beast/issues/1662
556  // DCHECK_EQ(sz_read, http_read) << " " << range.size() << "/" << left_available;
557  handler->offset += http_read;
558  return http_read;
559  }
560 
561  if (ec == asio::ssl::error::stream_truncated) {
562  LOG(WARNING) << "Stream " << handler->read_obj_url << " truncated at " << handler->offset
563  << "/" << handler->file_size;
564 
565  auto req = PrepareRequest(h2::verb::get, handler->read_obj_url, access_token_header_);
566  SetRange(handler->offset, kuint64max, &req);
567  OpenSeqResult res = OpenSequentialInternal(&req, &handler->parser);
568 
569  if (!res.ok())
570  return res.status;
571  VLOG(1) << "Reopened the file, new size: " << handler->offset + res.obj;
572  // I do not change seq_file_->offset,file_size fields.
573  // TODO: to validate that file version has not been changed between retries.
574  continue;
575  } else {
576  LOG(ERROR) << "ec: " << ec << "/" << ec.message() << " at " << handler->offset << "/"
577  << handler->file_size;
578  LOG(ERROR) << "FiberSocket status: " << https_client_->client()->next_layer().status();
579 
580  return ToStatus(ec);
581  }
582  }
583 
584  return Status(StatusCode::INTERNAL_ERROR, "Maximum iterations reached");
585 }

603 util::Status GCS::OpenForWrite(absl::string_view bucket, absl::string_view obj_path) {
604  RETURN_IF_ERROR(PrepareConnection());
605 
606  CHECK(absl::holds_alternative<absl::monostate>(*conn_state_));
607 
608  std::call_once(gcs_write_set_flag, [] {
609  gcs_writes.reset(new VarzQps("gcs-writes"));
610  gcs_latency.reset(new VarzMapAverage5m("gcs-latency"));
611  });
612 
613  string url = "/upload/storage/v1/b/";
614  absl::StrAppend(&url, bucket, "/o?uploadType=resumable&name=");
615  strings::AppendEncodedUrl(obj_path, &url);
616 
617  auto req = PrepareRequest(h2::verb::post, url, access_token_header_);
618  h2::response<h2::dynamic_body> resp_msg;
619 
620  req.prepare_payload();
621 
622  RETURN_IF_ERROR(HttpMessage(&req, &resp_msg));
623  if (resp_msg.result() != h2::status::ok) {
624  return HttpError(resp_msg);
625  }
626  auto it = resp_msg.find(h2::field::location);
627  if (it == resp_msg.end()) {
628  return Status(StatusCode::PARSE_ERROR, "Can not find location header");
629  }
630  CHECK_GE(FLAGS_gcs_upload_buf_log_size, 18);
631 
632  WriteHandler& wh = conn_state_->emplace<WriteHandler>();
633  wh.url = string(it->value());

640 util::Status GCS::Write(strings::ByteRange src) {
641  CHECK(!src.empty());
642  WriteHandler* wh = absl::get_if<WriteHandler>(conn_state_.get());
643  CHECK(wh && !wh->url.empty());
644  CHECK(io_context_.InContextThread());
645 
646  while (wh->Append(&src)) {
647  h2::request<h2::dynamic_body> req(h2::verb::put, wh->url, 11);
648 
649  size_t body_size = wh->body_mb.size();
650  size_t to = wh->uploaded + body_size;
651 
652  req.body() = std::move(wh->body_mb);
653  SetContentRange(wh->uploaded, to, 0, &req);
654  req.set(h2::field::content_type, "application/octet-stream");
655  req.prepare_payload();
656 
657  h2::response<h2::dynamic_body> resp_msg;
658  error_code ec;
659  unsigned retry = 0;
660  uint64_t start = base::GetMonotonicMicrosFast();
661 
662  for (; retry < 3; ++retry) {
663  VLOG(1) << "UploadReq" << retry << ": " << req << " socket "
664  << native_handle();
665  ec = https_client_->Send(req, &resp_msg);
666  if (ec) {
667  LOG(WARNING) << "retrying " << retry<< " after " << ec << "/" << ec.message();
668  gcs_latency->IncBy("retry", base::GetMonotonicMicrosFast() - start);
669  continue;
670  }
671 
672  VLOG(1) << "UploadResp" << retry << ": " << resp_msg;
673 
674  if (h2::status::permanent_redirect == resp_msg.result())
675  break;
676 
677  h2::status_class st_class = h2::to_status_class(resp_msg.result());
678  if (st_class == h2::status_class::server_error) {
679  VLOG(1) << "Retrying the service "<< retry;
680  this_fiber::sleep_for(10ms);
681  https_client_->schedule_reconnect();
682  gcs_latency->IncBy("retry", base::GetMonotonicMicrosFast() - start);
683  continue;
684  }
685  LOG(FATAL) << "Unexpected response: " << resp_msg;
686  }
687  RETURN_EC_STATUS(ec);
688  if (h2::status::service_unavailable == resp_msg.result()) {
689  LOG(FATAL) << "Service unavailable " << resp_msg << " after " << retry << " retries";
690  }
691 
692  auto it = resp_msg.find(h2::field::range);
693  CHECK(it != resp_msg.end()) << resp_msg;
694  absl::string_view range = absl_sv(it->value());
695  CHECK(absl::ConsumePrefix(&range, "bytes="));
696  size_t pos = range.find('-');
697  CHECK_LT(pos, range.size());
698  size_t uploaded_pos = 0;
699  CHECK(absl::SimpleAtoi(range.substr(pos + 1), &uploaded_pos));
700  CHECK_EQ(uploaded_pos + 1, to);
701 
702  // wh->body_mb = std::move(req.body());
703  // wh->body_mb.consume(body_size);
704  wh->uploaded = to;
705  CHECK_EQ(0, wh->body_mb.size());
706  // CHECK_EQ(wh->body_mb.capacity(), wh->body_mb.max_size());
707 
708  gcs_writes->Inc();
709  gcs_latency->IncBy("write", base::GetMonotonicMicrosFast() - start);
710 
711  if (src.empty())
712  break;
713  }
714 
715  return Status::OK;
